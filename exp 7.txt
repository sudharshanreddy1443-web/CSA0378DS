#include <stdio.h>

int main() {
    int arr[100], n, pos, item, choice;

    printf("Enter size of array: ");
    scanf("%d",&n);

    printf("Enter elements:\n");
    for(int i=0;i<n;i++)
        scanf("%d",&arr[i]);

    printf("\n1.Insert\n2.Delete\n3.Display\n");
    scanf("%d",&choice);

    switch(choice) {
        case 1:
            printf("Enter position and element: ");
            scanf("%d%d",&pos,&item);

            for(int i=n;i>=pos;i--)
                arr[i]=arr[i-1];

            arr[pos-1]=item;
            n++;
            break;

        case 2:
            printf("Enter position to delete: ");
            scanf("%d",&pos);

            for(int i=pos-1;i<n-1;i++)
                arr[i]=arr[i+1];

            n--;
            break;

        case 3:
            break;
    }

    printf("Array Elements:\n");
    for(int i=0;i<n;i++)
        printf("%d ",arr[i]);

    return 0;
}
